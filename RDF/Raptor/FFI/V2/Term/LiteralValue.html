def to_rdf
  str = self.to_str
  case
    when language = self.language
      RDF::Literal.new(str, language: language)
    when datatype = self.datatype
      RDF::Literal.new(str, datatype: datatype)
    else
      RDF::Literal.new(str)
  end
end
